#백준 체스판 다시 칠하기


![alt text](./assets/images/image1018.png)
완전 탐색 문제 풀이가 취약하단 생각이 들어서 기초 문제부터 천천히 풀어보기로 했다.

```
#include <iostream>
#include <vector>
using namespace std;
#define boardSize 8
#define INF 987654321
int returnCount(vector<string>& board, pair<int, int> startPos)
{
	int a_count = 0; int b_count = 0;
	for (int i = 0; i < boardSize; ++i)
	{
		for (int j = 0; j < boardSize; ++j)
		{
			int y = startPos.first + i; int x = startPos.second + j;
			if (board[y][x] == 'W')
			{
				if (y % 2 == x % 2) a_count++;
				else b_count++;
			}
			else
			{
				if (y % 2 == x % 2) b_count++;
				else a_count++;
			}
		}
	}
	return min(a_count, b_count);
}

int main()
{
	int n, m; cin >> n >> m;
	int answer = INF;
	vector<string> board(n);
	for (int i = 0; i < n; ++i) cin >> board[i];
	for (int i = 0; i + boardSize <= n; ++i)
	{
		for (int j = 0; j + boardSize <= m; ++j)
		{
			answer = min(answer, returnCount(board, make_pair(i, j)));
		}
	}
	cout << answer;
}
```
